CREATE TABLE JUDGE(JID NUMBER(10)PRIMARY KEY,JNAME VARCHAR2(20),J_PHONENO NUMBER(10));

CREATE TABLE PROSECUTOR(PID NUMBER(10)PRIMARY KEY,PNAME VARCHAR2(20),P_PHONENO NUMBER(10),EMAIL_ID VARCHAR2(20));

CREATE TABLE CRIME(CID NUMBER(10) PRIMARY KEY,TYPE VARCHAR2(10),LOCATION VARCHAR2(10),DATE DATE);

CREATE TABLE CRIMINAL(CRID NUMBER(10) PRIMARY KEY, CNAME VARCHAR2(10), CID NUMBER(10) REFERENCES CRIME(CID));

CREATE TABLE REGISTER(RID NUMBER(10) PRIMARY KEY,RNAME VARCHAR2(20));

CREATE TABLE CASE(CIN NUMBER(10),STARTINGDATE DATE,ENDINGDATE DATE,PID NUMBER(10) REFERENCES PROSECUTOR(PID),JID NUMBER(10) REFERENCES JUDGE(JID));

CREATE TABLE DEPENDENT(DNAME VARCHAR2(20)) PRIMARY KEY,PHONENO NUMBER(10),CIN NUMBER(10) REFERENCES CASE(CIN)

CREATE TABLE ADDRESS(DNAME VARCHAR2(20) REFERENCES DEPENDENT(DNAME),PHONENO NUMBER(10),DOORNO NUMBER(3),STREET VARCHAR2(15),CITY VARCHAR2(10),PINCODE NUMBER(6));

CREATE TABLE  RESCHEDULING(CIN NUMBER(10) REFERENCES CASE(CIN),RID NUMBER(10) REFERENCES REGISTER(RID),REASON VARCHAR2(10),PRIVIOUSDATE DATE,NEWDATE DATE);

CREATE TABLE ARRESTINGOFFICER (ANAME VARCHAR2(20),CRID NUMBER(10) REFERENCES CRIMINAL (CRID),DOARRESTING DATE);
